<script>
(() => {
    const root = document.querySelector('#dungeon-cards');
    const gridEl = root.querySelector('#dc-grid');
    const hpEl = root.querySelector('#dc-hp');
    const maxHpEl = root.querySelector('#dc-maxhp');
    const weaponEl = root.querySelector('#dc-weapon');
    const scoreEl = root.querySelector('#dc-score');
    const highEl = root.querySelector('#dc-high');
    const overlay = root.querySelector('#dc-overlay');
    const startBtn = root.querySelector('#dc-start-btn');
    const getPlayCountEl = () => document.querySelector('[data-aomagame-play-count]');
    const PLAYED_KEY = 'aomagame:played:dungeon-cards';

    // Game Config
    const GRID_W = 3;
    const GRID_H = 3;
    
    // State
    let hp = 10;
    let maxHp = 10;
    let weapon = 0;
    let score = 0;
    let highScore = 0;
    
    // Grid: Array of Cards
    // Card Types: 'player', 'enemy', 'potion', 'weapon', 'coin'
    let grid = [];
    let playerPos = 4; // Center (0-8)

    // Audio
    let audioCtx = null;
    const ensureAudio = () => {
        const C = window.AudioContext || window.webkitAudioContext;
        if (!C) return;
        if (!audioCtx) audioCtx = new C();
        if (audioCtx.state === 'suspended') audioCtx.resume().catch(()=>{});
    };

    function playSound(type) {
        if (!audioCtx) return;
        const now = audioCtx.currentTime;
        const osc = audioCtx.createOscillator();
        const gain = audioCtx.createGain();
        osc.connect(gain);
        gain.connect(audioCtx.destination);
        
        if (type === 'move') {
            osc.frequency.setValueAtTime(300, now);
            osc.frequency.exponentialRampToValueAtTime(50, now+0.1);
            gain.gain.setValueAtTime(0.1, now);
            gain.gain.linearRampToValueAtTime(0, now+0.1);
            osc.start(now);
            osc.stop(now+0.1);
        } else if (type === 'hit') {
            osc.type = 'square';
            osc.frequency.setValueAtTime(150, now);
            osc.frequency.linearRampToValueAtTime(100, now+0.1);
            gain.gain.setValueAtTime(0.1, now);
            gain.gain.linearRampToValueAtTime(0, now+0.1);
            osc.start(now);
            osc.stop(now+0.1);
        } else if (type === 'item') {
            osc.type = 'sine';
            osc.frequency.setValueAtTime(600, now);
            osc.frequency.linearRampToValueAtTime(1200, now+0.2);
            gain.gain.setValueAtTime(0.1, now);
            gain.gain.linearRampToValueAtTime(0, now+0.2);
            osc.start(now);
            osc.stop(now+0.2);
        } else if (type === 'gameover') {
            osc.type = 'sawtooth';
            osc.frequency.setValueAtTime(200, now);
            osc.frequency.linearRampToValueAtTime(50, now+1.0);
            gain.gain.setValueAtTime(0.2, now);
            gain.gain.linearRampToValueAtTime(0, now+1.0);
            osc.start(now);
            osc.stop(now+1.0);
        }
    }

    function init() {
        startBtn.addEventListener('click', startGame);
        
        // Restore high score?
        // simple session high score for now
        
        renderGrid();
    }

    function startGame() {
        ensureAudio();
        hp = 10;
        maxHp = 10;
        weapon = 0;
        score = 0;
        playerPos = 4; // Center
        
        grid = [];
        for(let i=0; i<9; i++) {
            if (i === playerPos) {
                grid.push({ type: 'player', val: 0 });
            } else {
                grid.push(generateCard());
            }
        }
        
        updateUI();
        overlay.classList.add('hidden');
        markPlayed();
    }
    
    function generateCard() {
        // Probabilities
        const r = Math.random();
        if (r < 0.4) {
            // Enemy
            // Strength based on score?
            const str = Math.floor(Math.random() * 5) + 2 + Math.floor(score/50); 
            return { type: 'enemy', val: str };
        } else if (r < 0.6) {
            // Potion
            const heal = Math.floor(Math.random() * 3) + 2; 
            return { type: 'potion', val: heal };
        } else if (r < 0.75) {
            // Weapon
            const str = Math.floor(Math.random() * 3) + 3;
            return { type: 'weapon', val: str };
        } else {
            // Coin
            const val = Math.floor(Math.random() * 5) + 10;
            return { type: 'coin', val: val };
        }
    }

    function movePlayer(idx) {
        if (hp <= 0) return;
        
        // Check adjacency
        const px = playerPos % 3;
        const py = Math.floor(playerPos / 3);
        const tx = idx % 3;
        const ty = Math.floor(idx / 3);
        
        // Dist must be 1 (Manhattan)
        if (Math.abs(px - tx) + Math.abs(py - ty) !== 1) return;
        
        const card = grid[idx];
        let moved = false;
        
        // Interact
        if (card.type === 'enemy') {
            // Combat
            let dmg = card.val;
            
            // Use weapon
            if (weapon > 0) {
                const reduce = Math.min(weapon, dmg);
                weapon -= reduce;
                dmg -= reduce;
                playSound('hit');
            }
            
            if (dmg > 0) {
                hp -= dmg;
                playSound('hit');
            } else {
                playSound('move');
            }
            
            // Enemy Defeated?
            // In this simple version, enemies are always defeated/removed when you step on them
            // But if you took remaining damage.
            
            // Score for defeating
            score += card.val * 2;
            moved = true;
            
        } else if (card.type === 'potion') {
            hp = Math.min(maxHp, hp + card.val);
            playSound('item');
            moved = true;
        } else if (card.type === 'weapon') {
            weapon += card.val; // Stack? or Replace? Stack is more fun
            playSound('item');
            moved = true;
        } else if (card.type === 'coin') {
            score += card.val;
            playSound('item');
            moved = true;
        }
        
        if (moved) {
            // Update Grid
            // Old player pos gets new card
            grid[playerPos] = generateCard();
            // New pos becomes player
            grid[idx] = { type: 'player', val: 0 };
            playerPos = idx;
            
            if (hp <= 0) {
                gameOver();
            } else {
                updateUI();
            }
        }
    }
    
    function gameOver() {
        hp = 0;
        updateUI();
        playSound('gameover');
        
        if (score > highScore) {
            highScore = score;
        }
        
        overlay.classList.remove('hidden');
        root.querySelector('h2').innerText = "GAME OVER";
        root.querySelector('p').innerHTML = `SCORE: ${score}<br>HIGH SCORE: ${highScore}`;
        startBtn.innerText = "RETRY";
    }

    function updateUI() {
        hpEl.innerText = hp;
        
        // Max HP grows? Maybe every 100 score?
        // Let's implement simple scaling
        const scaledMax = 10 + Math.floor(score / 200) * 2;
        if (scaledMax > maxHp) maxHp = scaledMax;
        
        maxHpEl.innerText = maxHp;
        weaponEl.innerText = weapon;
        scoreEl.innerText = score;
        highEl.innerText = highScore;
        
        renderGrid();
    }

    function renderGrid() {
        gridEl.innerHTML = '';
        
        grid.forEach((card, idx) => {
            const div = document.createElement('div');
            
            // Style
            div.className = `card ${card.type}`;
            
            // Icon
            let icon = '';
            let val = '';
            if (card.type === 'player') {
                icon = 'ğŸ§™â€â™‚ï¸';
            } else if (card.type === 'enemy') {
                icon = 'ğŸ‘¿';
                val = card.val;
            } else if (card.type === 'potion') {
                icon = 'ğŸ·';
                val = '+' + card.val;
            } else if (card.type === 'weapon') {
                icon = 'âš”ï¸';
                val = card.val;
            } else if (card.type === 'coin') {
                icon = 'ğŸ’°';
                val = card.val;
            }
            
            div.innerHTML = `<div class="card-icon">${icon}</div><div class="card-val">${val}</div>`;
            
            // Check adjacency for disabling non-movable cards
            if (hp > 0 && grid.length > 0) { // If playing
                const px = playerPos % 3;
                const py = Math.floor(playerPos / 3);
                const tx = idx % 3;
                const ty = Math.floor(idx / 3);
                const dist = Math.abs(px - tx) + Math.abs(py - ty);
                
                if (dist !== 1 && idx !== playerPos) {
                   div.classList.add('disabled');
                } else {
                   // Active
                   if (idx !== playerPos) {
                       div.onclick = () => movePlayer(idx);
                   }
                }
            } else {
                // Not playing or game over
                div.classList.add('disabled');
            }

            gridEl.appendChild(div);
        });
    }

    // Play Count Logic
    function updatePlayCount() {
        const counterEl = getPlayCountEl();
        if (!counterEl) return;
        try {
            let total = 0;
            for (let i = 0; i < localStorage.length; i++) {
                const key = localStorage.key(i);
                if (key && key.startsWith('aomagame:played:')) {
                     const val = parseInt(localStorage.getItem(key)||'0', 10);
                     if (val > 0) total++;
                }
            }
            counterEl.textContent = total;
        } catch (e) { counterEl.textContent = '0'; }
    }

    function markPlayed() {
        try {
            const current = parseInt(localStorage.getItem(PLAYED_KEY) || '0', 10);
            localStorage.setItem(PLAYED_KEY, String(current + 1));
        } catch(e) {}
        updatePlayCount();
    }

    init();
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', updatePlayCount, { once: true });
    } else {
        updatePlayCount();
    }
})();
</script>
